# In case the swarm doesn't work properly, use these commands to do a full reset(not recommended)
docker ps --format '{{.Names}}' | grep -v 'keycloak' | xargs -r docker stop
docker ps -a --format '{{.Names}}' | grep -v 'keycloak' | xargs -r docker rm
docker images --format '{{.Repository}}:{{.Tag}}' | grep -v 'keycloak' | xargs -r docker rmi
docker volume prune -f
docker network ls --format '{{.Name}}' | grep -v -E 'bridge|host|none' | xargs -r docker network rm
